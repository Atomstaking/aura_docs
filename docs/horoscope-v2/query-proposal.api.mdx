## Query proposal

Query: 
```graphql
query queryProposal(
  $limit: Int = 10
  $offset: Int = 0
  $order: order_by = desc
  $proposalId: Int = null
  $type: String = null
) {
  xstaxy {
    proposal(
      limit: $limit
      offset: $offset
      where: { proposal_id: { _eq: $proposalId }, type: { _eq: $type } }
      order_by: { proposal_id: $order }
    ) {
      content
      deposit_end_time
      description
      initial_deposit
      proposal_id
      proposer_address
      count_vote
      proposer {
        description
        operator_address
        account_address
      }
      status
      submit_time
      tally
      title
      total_deposit
      turnout
      type
      updated_at
      voting_end_time
      voting_start_time
    }
    proposal_aggregate(
      where: { proposal_id: { _eq: $proposalId }, type: { _eq: $type } }
    ) {
      aggregate {
        count
      }
    }
  }
}
```
